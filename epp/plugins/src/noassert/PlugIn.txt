/*
 Usage:
	epp -plug-in noassert File1.java File2.java ...

 This plug-in can override the effect of #epp "assert".

*/

#epp "Symbol"
#epp "SystemMixin"
#epp "AutoSplitFiles"
#epp "BackQuote"


package noassert;
import epp.*;

SystemMixin noassert {
  class Epp {

    extend void initInvokeStyleMacroTable() {
      original();
      redefineInvokeStyleMacro(:assert, new NoassertMacro());
    }
  }
}

class NoassertMacro extends InvokeStyleMacro {
  
  public Tree call(Tree tree, Tree[] args){
    return `(emptyStatement);
  }
}
